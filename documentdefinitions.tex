\starttexdefinition see [#1:#2]
  \doifdefinedelse{in#1}
    {\getvalue{in#1}[#1:#2]}
    {\writestatus{warning}{referenceformat in#1 not defined}
     \in[#1:#2]}
\stoptexdefinition

   right=\vert]

   right=\rangle]

  [left=\langle,
   right=\rangle]
