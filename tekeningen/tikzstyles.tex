
% Be aware of the space in the definitions below!
\def\setuptikzstyle[#1] [#2]
  {\tikzset{#1/.style={#2}}}
\def\letcolor[#1] [#2]
  {\colorlet{#1}{#2}}

\def\Gauss#1#2% mu sigma
  {1/(#2 * sqrt(2*pi)) * exp(-1/2 * ((\x - #1)/#2)^2)}
\def\Diffraction
  {(cos(10*pi * sin(\x/10)))^2}
\def\Inference#1% N
  {((sin(#1 * \Displacement))/(\Displacement))^2}
\def\Displacement{2*pi * sin(\x/10)}

\letcolor[light orange]
  [orange!33]
\letcolor[background orange]
  [orange!10]

% Every:
\setuptikzstyle[every picture]
  [>=stealth,font=\small]
\setuptikzstyle[every node]
  [black]

% Paths:

  [->,thick,gray]
\setuptikzstyle[screen lines]

\setuptikzstyle[just lines]
  [orange]
\setuptikzstyle[light lines]
  [light orange]
\setuptikzstyle[heavy lines]
  [very thick,orange]
\setuptikzstyle[slit]
  [white,radius=0.5]
